46 /**
47  * @brief Distance Detector result returned from @ref acc_detector_distance_process
48  */
49 typedef struct
50 {
51  /**
52  * Estimated distances (m).
53  *
54  * Sorted according to the selected peak sorting strategy.
55  */
56  float distances[ACC_DETECTOR_DISTANCE_RESULT_MAX_NUM_DISTANCES];
57 
58  /**
59  * Estimated reflector strengths (dB).
60  *
61  * Corresponds to the peak amplitude of the estimated distances.
62  */
63  float strengths[ACC_DETECTOR_DISTANCE_RESULT_MAX_NUM_DISTANCES];
64 
65  /**
66  * The number of valid distances in the distances array.
67  *
68  * Corresponds to the number of detected objects according to the threshold method.
69  * If 0, no objects were detected.
70  */
71  uint8_t num_distances;
72 
73  /**
74  * Indication of an object close to the start edge, located outside of the measurement range.
75  */
76  bool near_start_edge_status;
77 
78  /**
79  * Indication of calibration needed.
80  *
81  * Is set if the temperature has changed by at least 15 degrees Celsius from
82  * the temperature at calibration.
83  *
84  * If true:
85  * - The sensor calibration needs to be redone by calling @ref acc_sensor_calibrate
86  * - The Detector calibration needs to be updated by calling @ref acc_detector_distance_update_calibration
87  */
88  bool calibration_needed;
89 
90  /**
91  * Temperature in sensor during measurement (in degree Celsius).
92  *
93  * Note that it has poor absolute accuracy and should only be used
94  * for relative temperature measurements.
95  */
96  int16_t temperature;
97 
98  /**
99  * Processing result, including IQ data, that the Distance detection is based on.
100  *
101  * This will point to memory in the buffer supplied to @ref acc_detector_distance_process
102  *
103  * Note: The processing result is only valid until the next time
104  * @ref acc_detector_distance_process is called.
105  */
106  acc_processing_result_t *processing_result;
107 
108  /**
109  * Metadata for the processing result
110  *
111  * Note: The processing metedata is only valid until the next time
112  * @ref acc_detector_distance_process is called.
113  */
114  acc_processing_metadata_t *processing_metadata;
115 
116  /**
117  * The sensor_config used for the processing result
118  *
119  * Note: The sensor_config is only valid until the next time
120  * @ref acc_detector_distance_process is called.
121  */
122  const acc_config_t *sensor_config;
123 } acc_detector_distance_result_t;

346 /**
347  * @brief Set maximum step length
348  *
349  * Step length is the number of 2.5 mm distance points between each
350  * measurement point in the underlying IQ data. This is normally
351  * automatically set by the detector depending on the range start/end
352  * configurations.
353  *

362  * @param[in] max_step_length The maximum step length
363  */
