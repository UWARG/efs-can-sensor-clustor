68 {
69  GPIO_PinState sel0 = GPIO_PIN_RESET;
70  GPIO_PinState sel1 = GPIO_PIN_RESET;
71  GPIO_PinState sel2 = GPIO_PIN_RESET;
72 
73  switch (sensor_id)
74  {
75  case 1:
76  break;
77  case 2:
78  sel0 = GPIO_PIN_SET;
79  break;
80  case 3:
81  sel1 = GPIO_PIN_SET;
82  break;
83  case 4:
84  sel0 = GPIO_PIN_SET;
85  sel1 = GPIO_PIN_SET;
86  break;
87  case 5:
88  sel2 = GPIO_PIN_SET;
89  break;
90  default:
91  break;
92  }
93 
94  HAL_GPIO_WritePin(SPI_SEL0_GPIO_Port, SPI_SEL0_Pin, sel0);
95  HAL_GPIO_WritePin(SPI_SEL1_GPIO_Port, SPI_SEL1_Pin, sel1);
96  HAL_GPIO_WritePin(SPI_SEL2_GPIO_Port, SPI_SEL2_Pin, sel2);

99 static void acc_hal_integration_sensor_transfer(acc_sensor_id_t sensor_id, uint8_t *buffer, size_t buffer_size)
100 {
101  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
102  {
103  Error_Handler();
104  }
105 
106  spi_select(sensor_id);
107 
108  const uint32_t SPI_TRANSMIT_RECEIVE_TIMEOUT = 5000;
109 
110  HAL_GPIO_WritePin(A121_SPI_SS_GPIO_Port, A121_SPI_SS_Pin, GPIO_PIN_RESET);
111 
112 #ifdef STM32_USE_SPI_DMA
113  spi_transfer_complete = false;
114  HAL_StatusTypeDef status = HAL_SPI_TransmitReceive_DMA(&A121_SPI_HANDLE, spi_buffer, spi_buffer, buffer_size);
115 
116  if (status != HAL_OK)
117  {
118  Error_Handler();
119  }
120 
121  uint32_t start = HAL_GetTick();
122 
123  while (!spi_transfer_complete && (HAL_GetTick() - start) < SPI_TRANSMIT_RECEIVE_TIMEOUT)
124  {
125  // Turn off interrupts
126  disable_interrupts();
127  // Check once more so that the interrupt have not occurred
128  if (!spi_transfer_complete)
129  {
130  __WFI();
131  }
132 
133  // Enable interrupt again, the ISR will execute directly after this
134  enable_interrupts();
135  }
136 
137 #else
138  HAL_SPI_TransmitReceive(&A121_SPI_HANDLE, buffer, buffer, buffer_size, SPI_TRANSMIT_RECEIVE_TIMEOUT);
139 #endif
140 
141  HAL_GPIO_WritePin(A121_SPI_SS_GPIO_Port, A121_SPI_SS_Pin, GPIO_PIN_SET);
142 }

164 void acc_hal_integration_sensor_enable(acc_sensor_id_t sensor_id)
165 {
166  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
167  {
168  Error_Handler();
169  }
170 
171  const acc_sensor_info_t *sensor_info = &sensor_infos[sensor_id - 1];
172 
173  HAL_GPIO_WritePin(sensor_info->en_port, sensor_info->en_pin_mask, GPIO_PIN_SET);
174 
175  // Wait 2 ms to make sure that the sensor crystal has time to stabilize
176  acc_integration_sleep_us(2000);
177 }

195 bool acc_hal_integration_wait_for_sensor_interrupt(acc_sensor_id_t sensor_id, uint32_t timeout_ms)
196 {
197  bool status = false;
198 
199  if ((sensor_id == 0) || (sensor_id > SENSOR_COUNT))
200  {
201  Error_Handler();
202  }
203 
204  const acc_sensor_info_t *sensor_info = &sensor_infos[sensor_id - 1];
205  GPIO_TypeDef *int_port = sensor_info->int_port;
206  uint32_t int_pin_mask = sensor_info->int_pin_mask;
207  const uint32_t wait_begin_ms = HAL_GetTick();
208 
209  while ((HAL_GPIO_ReadPin(int_port, int_pin_mask) != GPIO_PIN_SET) && (HAL_GetTick() - wait_begin_ms < timeout_ms))
210  {
211  // Wait for the GPIO interrupt
212  disable_interrupts();
213  // Check again so that IRQ did not occur
214  if (HAL_GPIO_ReadPin(int_port, int_pin_mask) != GPIO_PIN_SET)
215  {
216  __WFI();
217  }
218 
219  // Enable interrupts again to allow pending interrupt to be handled
220  enable_interrupts();
221  }
222 
223  status = HAL_GPIO_ReadPin(int_port, int_pin_mask) == GPIO_PIN_SET;
224 
225  return status;
226 }
