135 hand_motion_detection_handle_t *hand_motion_detection_handle_create(const hand_motion_detection_config_t *hand_motion_detection_config,
136  acc_sensor_id_t sensor_id)
137 {
138  hand_motion_detection_handle_t *handle = acc_integration_mem_calloc(1U, sizeof(*handle));
139 
140  bool status = handle != NULL;
141 
142  if (status)
143  {
144  handle->config = *hand_motion_detection_config;
145  handle->prepare_needed = true;
146  handle->app_mode = HAND_MOTION_DETECTION_APP_MODE_HAND_MOTION;
147  handle->sensor_id = sensor_id;
148 
149  handle->hand_motion_detection_timeout_duration =
150  (uint16_t)(roundf(hand_motion_detection_config->hand_detection_timeout * hand_motion_detection_config->algo_config.frame_rate));
151  handle->hand_motion_detection_timer = 0U;
152  handle->detection_retention_duration = (uint16_t)(roundf(hand_motion_detection_config->algo_config.detection_retention_duration *
153  hand_motion_detection_config->algo_config.frame_rate));
154 
155  reinit_hand_motion_detection_state(handle);
156 
157  handle->hand_motion_detection_sensor_config = acc_config_create();
158  handle->hand_motion_detection_presence_processing_config = acc_detector_presence_processing_config_create();
159 
160  status = handle->hand_motion_detection_sensor_config != NULL && handle->hand_motion_detection_presence_processing_config != NULL;
161  }
162 
163  if (status)
164  {
165  set_sensor_config(hand_motion_detection_config, handle->hand_motion_detection_sensor_config);
166  set_processing_config(hand_motion_detection_config, handle->hand_motion_detection_presence_processing_config);
167 
168  handle->sensor_processing = acc_processing_create(handle->hand_motion_detection_sensor_config, &handle->sensor_processing_metadata);
169  handle->hand_motion_detection_presence_processor = acc_detector_presence_processing_create(
170  handle->hand_motion_detection_presence_processing_config, handle->hand_motion_detection_sensor_config);
171 
172  status = handle->hand_motion_detection_presence_processor != NULL;
173  }
174 
175  if (status)
176  {
177  if (handle->config.use_presence_detection)
178  {
179  handle->app_mode = HAND_MOTION_DETECTION_APP_MODE_PRESENCE;
180 
181  handle->presence_detector = acc_detector_presence_create(handle->config.presence_config, &handle->presence_metadata);
182 
183  status = handle->presence_detector != NULL;
184  }
185  }
186 
187  if (!status)
188  {
189  hand_motion_detection_handle_destroy(handle);
190  handle = NULL;
191  }
192 
193  return handle;
194 }

312 {

350 static void set_sensor_config(const hand_motion_detection_config_t *hand_motion_detection_config, acc_config_t *sensor_config)
351 {
352  const hand_motion_detection_algo_config_t *algo_config = &hand_motion_detection_config->algo_config;
353  acc_config_profile_t profile = ACC_CONFIG_PROFILE_1;
354  uint16_t step_length = 6U;
355  float water_jet_half_width = algo_config->water_jet_width / 2.0f;
356  uint8_t subsweep_idx = 0U;
357  float profile_envelope_fwhm_m = acc_algorithm_get_fwhm(profile);
358 
359  // Determine if subsweep between sensor and water jet is feasible
360  if (profile_envelope_fwhm_m < (algo_config->sensor_to_water_distance - water_jet_half_width))
361  {
362  int32_t start_point =
363  (int32_t)roundf((algo_config->sensor_to_water_distance - profile_envelope_fwhm_m - water_jet_half_width) / ACC_APPROX_BASE_STEP_LENGTH_M);
364  acc_config_subsweep_start_point_set(sensor_config, start_point, subsweep_idx);
365  acc_config_subsweep_num_points_set(sensor_config, 1U, subsweep_idx);
366  acc_config_subsweep_step_length_set(sensor_config, step_length, subsweep_idx);
367  acc_config_subsweep_profile_set(sensor_config, profile, subsweep_idx);
368  acc_config_subsweep_hwaas_set(sensor_config, algo_config->hwaas, subsweep_idx);
369  acc_config_subsweep_receiver_gain_set(sensor_config, 4U, subsweep_idx);
370 
371  subsweep_idx++;
372  }
373 
374  // Determine if subsweep after water jet is required
375  if (profile_envelope_fwhm_m + water_jet_half_width + algo_config->sensor_to_water_distance < algo_config->measurement_range_end)
376  {
377  float start_m = algo_config->sensor_to_water_distance + water_jet_half_width + profile_envelope_fwhm_m;
378  int32_t start_point = (int32_t)roundf(start_m / ACC_APPROX_BASE_STEP_LENGTH_M);
379  uint16_t num_points =
380  (uint16_t)roundf(fmaxf(1.0f, (algo_config->measurement_range_end - start_m) / ((float)step_length * ACC_APPROX_BASE_STEP_LENGTH_M)));
381 
382  acc_config_subsweep_start_point_set(sensor_config, start_point, subsweep_idx);
383  acc_config_subsweep_num_points_set(sensor_config, num_points, subsweep_idx);
384  acc_config_subsweep_step_length_set(sensor_config, step_length, subsweep_idx);
385  acc_config_subsweep_profile_set(sensor_config, profile, subsweep_idx);
386  acc_config_subsweep_hwaas_set(sensor_config, algo_config->hwaas, subsweep_idx);
387  acc_config_subsweep_receiver_gain_set(sensor_config, 4U, subsweep_idx);
388 
389  subsweep_idx++;
390  }
391 
392  acc_config_num_subsweeps_set(sensor_config, subsweep_idx);
393  acc_config_frame_rate_set(sensor_config, algo_config->frame_rate);
394  acc_config_sweep_rate_set(sensor_config, algo_config->sweep_rate);
395  acc_config_sweeps_per_frame_set(sensor_config, algo_config->sweeps_per_frame);
396  acc_config_inter_frame_idle_state_set(sensor_config, ACC_CONFIG_IDLE_STATE_SLEEP);
397  acc_config_inter_sweep_idle_state_set(sensor_config, ACC_CONFIG_IDLE_STATE_SLEEP);
398 }

420 static hand_motion_detection_detection_state_t process_presence_processing_result(
421  hand_motion_detection_handle_t *handle,
422  const acc_detector_presence_processing_result_t *presence_processing_result)
423 {
424  hand_motion_detection_detection_state_t detection_state = HAND_MOTION_DETECTION_STATE_NO_DETECTION;
425 
426  if (presence_processing_result->presence_detected)
427  {
428  handle->has_detected = true;
429  handle->update_index_at_detection = handle->update_index;
430  detection_state = HAND_MOTION_DETECTION_STATE_DETECTION;
431  }
432 
433  if (handle->has_detected && detection_state != HAND_MOTION_DETECTION_STATE_DETECTION &&
434  ((handle->update_index - handle->update_index_at_detection) < handle->detection_retention_duration))
435  {
436  detection_state = HAND_MOTION_DETECTION_STATE_RETENTION;
437  }
438 
439  handle->update_index++;
440 
441  return detection_state;
442 }

444 static void determine_app_mode(hand_motion_detection_handle_t *handle, hand_motion_detection_result_t *result)
445 {
446  hand_motion_detection_app_mode_t new_app_mode = handle->app_mode;
447 
448  if (handle->app_mode == HAND_MOTION_DETECTION_APP_MODE_PRESENCE)
449  {
450  if (result->presence_result.presence_detected)
451  {
452  new_app_mode = HAND_MOTION_DETECTION_APP_MODE_HAND_MOTION;
453  }
454  }
455  else // HAND_MOTION_DETECTION_APP_MODE_HAND_MOTION
456  {
457  if (handle->config.use_presence_detection)
458  {
459  if (result->algo_result.detection_state != HAND_MOTION_DETECTION_STATE_NO_DETECTION)
460  {
461  handle->hand_motion_detection_timer = 0U;
462  }
463  else if (handle->hand_motion_detection_timeout_duration < handle->hand_motion_detection_timer)
464  {
465  new_app_mode = HAND_MOTION_DETECTION_APP_MODE_PRESENCE;
466  }
467  else
468  {
469  handle->hand_motion_detection_timer++;
470  }
471  }
472  }
473 
474  if (new_app_mode != handle->app_mode)
475  {
476  swap_app_mode(handle, new_app_mode);
477  }
478 }
