51 int acc_example_hand_motion_detection_main(int argc, char *argv[])
52 {
53  (void)argc;
54  (void)argv;
55  acc_sensor_t *sensor = NULL;
56  acc_cal_result_t cal_result;
57  void *buffer = NULL;
58  uint32_t buffer_size = 0;
59 
60  hand_motion_detection_config_t *hand_motion_detection_config = NULL;
61  hand_motion_detection_handle_t *hand_motion_detection_handle = NULL;
62  hand_motion_detection_result_t hand_motion_detection_result = {0};
63 
64  printf("Acconeer software version %s\n", acc_version_get());
65 
66  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
67 
68  if (!acc_rss_hal_register(hal))
69  {
70  return EXIT_FAILURE;
71  }
72 
73  hand_motion_detection_config = hand_motion_detection_config_create();
74 
75  if (hand_motion_detection_config == NULL)
76  {
77  printf("hand_motion_detection_config_create() failed\n");
78  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
79  return EXIT_FAILURE;
80  }
81 
82  hand_motion_detection_set_config(HAND_MOTION_DETECTION_PRESET_FAUCET, hand_motion_detection_config);
83 
84  hand_motion_detection_handle = hand_motion_detection_handle_create(hand_motion_detection_config, SENSOR_ID);
85  if (hand_motion_detection_handle == NULL)
86  {
87  printf("hand_motion_detection_handle_create() failed\n");
88  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
89  return EXIT_FAILURE;
90  }
91 
92  hand_motion_detection_config_log(hand_motion_detection_handle);
93 
94  if (!hand_motion_detection_get_buffer_size(hand_motion_detection_handle, &buffer_size))
95  {
96  printf("acc_rss_get_buffer_size() failed\n");
97  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
98  return EXIT_FAILURE;
99  }
100 
101  buffer = acc_integration_mem_alloc(buffer_size);
102  if (buffer == NULL)
103  {
104  printf("buffer allocation failed\n");
105  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
106  return EXIT_FAILURE;
107  }
108 
109  acc_hal_integration_sensor_supply_on(SENSOR_ID);
110  acc_hal_integration_sensor_enable(SENSOR_ID);
111 
112  sensor = acc_sensor_create(SENSOR_ID);
113  if (sensor == NULL)
114  {
115  printf("acc_sensor_create() failed\n");
116  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
117  return EXIT_FAILURE;
118  }
119 
120  if (!do_sensor_calibration(sensor, &cal_result, buffer, buffer_size))
121  {
122  printf("do_sensor_calibration() failed\n");
123  acc_sensor_status(sensor);
124  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
125  return EXIT_FAILURE;
126  }
127 
128  while (true)
129  {
130  if (!hand_motion_detection_prepare(hand_motion_detection_handle, sensor, &cal_result, buffer, buffer_size, false))
131  {
132  printf("hand_motion_detection_prepare() failed\n");
133  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
134  return EXIT_FAILURE;
135  }
136 
137  if (!acc_sensor_measure(sensor))
138  {
139  printf("acc_sensor_measure failed\n");
140  acc_sensor_status(sensor);
141  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
142  return EXIT_FAILURE;
143  }
144 
145  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
146  {
147  printf("Sensor interrupt timeout\n");
148  acc_sensor_status(sensor);
149  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
150  return EXIT_FAILURE;
151  }
152 
153  if (!acc_sensor_read(sensor, buffer, buffer_size))
154  {
155  printf("acc_sensor_read failed\n");
156  acc_sensor_status(sensor);
157  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
158  return EXIT_FAILURE;
159  }
160 
161  hand_motion_detection_process(hand_motion_detection_handle, buffer, &hand_motion_detection_result);
162 
163  bool data_reliable;
164 
165  if (!handle_indications(
166  hand_motion_detection_handle, &hand_motion_detection_result, sensor, &cal_result, buffer, buffer_size, &data_reliable))
167  {
168  printf("handle_indications() failed\n");
169  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
170  return EXIT_FAILURE;
171  }
172 
173  if (data_reliable)
174  {
175  print_hand_motion_detection_result(&hand_motion_detection_result);
176  }
177  }
178 
179  cleanup(sensor, buffer, hand_motion_detection_config, hand_motion_detection_handle);
180 
181  printf("Application finished OK\n");
182 
183  return EXIT_SUCCESS;
184 }

186 static void cleanup(acc_sensor_t *sensor, void *buffer, hand_motion_detection_config_t *config, hand_motion_detection_handle_t *handle)
187 {
188  acc_hal_integration_sensor_disable(SENSOR_ID);
189  acc_hal_integration_sensor_supply_off(SENSOR_ID);
190 
191  if (sensor != NULL)
192  {
193  acc_sensor_destroy(sensor);
194  }
195 
196  if (buffer != NULL)
197  {
198  acc_integration_mem_free(buffer);
199  }
200 
201  if (config != NULL)
202  {
203  hand_motion_detection_config_destroy(config);
204  }
205 
206  if (handle != NULL)
207  {
208  hand_motion_detection_handle_destroy(handle);
209  }
210 }

212 static bool do_sensor_calibration(acc_sensor_t *sensor, acc_cal_result_t *cal_result, void *buffer, uint32_t buffer_size)
213 {
214  bool status = false;
215  bool cal_complete = false;
216  const uint16_t calibration_retries = 1U;
217 
218  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
219  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
220  {
221  // Reset sensor before calibration by disabling/enabling it
222  acc_hal_integration_sensor_disable(SENSOR_ID);
223  acc_hal_integration_sensor_enable(SENSOR_ID);
224 
225  do
226  {
227  status = acc_sensor_calibrate(sensor, &cal_complete, cal_result, buffer, buffer_size);
228 
229  if (status && !cal_complete)
230  {
231  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
232  }
233  } while (status && !cal_complete);
234  }
235 
236  if (status)
237  {
238  // Reset sensor after calibration by disabling/enabling it
239  acc_hal_integration_sensor_disable(SENSOR_ID);
240  acc_hal_integration_sensor_enable(SENSOR_ID);
241  }
242 
243  return status;
244 }

246 static bool handle_indications(hand_motion_detection_handle_t *handle,
247  hand_motion_detection_result_t *result,
248  acc_sensor_t *sensor,
249  acc_cal_result_t *cal_result,
250  void *buffer,
251  uint32_t buffer_size,
252  bool *data_reliable)
253 {
254  bool status = true;
255 
256  *data_reliable = true;
257 
258  if (result->proc_result.data_saturated)
259  {
260  *data_reliable = false;
261 
262  printf("Data saturated. Try to reduce the sensor gain.\n");
263  }
264 
265  if (result->proc_result.frame_delayed)
266  {
267  printf("Frame delayed. Could not read data fast enough.\n");
268  printf("Try lowering the frame rate or call 'acc_sensor_read' more frequently.\n");
269  }
270 
271  if (result->proc_result.calibration_needed)
272  {
273  printf("The current calibration is not valid for the current temperature.\n");
274  printf("Re-calibrating sensor...\n");
275 
276  if (status)
277  {
278  if (!do_sensor_calibration(sensor, cal_result, buffer, buffer_size))
279  {
280  printf("do_sensor_calibration() failed\n");
281  acc_sensor_status(sensor);
282  status = false;
283  }
284  else
285  {
286  printf("Sensor recalibration done!\n");
287  }
288  }
289 
290  if (status)
291  {
292  if (!hand_motion_detection_prepare(handle, sensor, cal_result, buffer, buffer_size, true))
293  {
294  printf("hand_motion_detection_prepare() failed\n");
295  status = false;
296  }
297  }
298 
299  *data_reliable = false;
300 
301  printf("The sensor was successfully re-calibrated.\n");
302  }
303 
304  return status;
305 }

307 static void print_hand_motion_detection_result(const hand_motion_detection_result_t *result)
308 {
309  printf("App mode: %s\n", result->app_mode == HAND_MOTION_DETECTION_APP_MODE_PRESENCE ? "presence" : "handmotion");
310 
311  if (result->algo_result_available)
312  {
313  printf("Detection state: ");
314 
315  switch (result->algo_result.detection_state)
316  {
317  case HAND_MOTION_DETECTION_STATE_NO_DETECTION:
318  printf("No detection\n");
319  break;
320  case HAND_MOTION_DETECTION_STATE_DETECTION:
321  printf("Detection\n");
322  break;
323  case HAND_MOTION_DETECTION_STATE_RETENTION:
324  printf("Retention\n");
325  break;
326  }
327  }
