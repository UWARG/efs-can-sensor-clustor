56 int acc_ref_app_breathing_main(int argc, char *argv[])
57 {
58  (void)argc;
59  (void)argv;
60  ref_app_breathing_config_t *config = NULL;
61  ref_app_breathing_handle_t *handle = NULL;
62  acc_sensor_t *sensor = NULL;
63  acc_cal_result_t sensor_cal_result;
64  void *buffer = NULL;
65  uint32_t buffer_size = 0U;
66  ref_app_breathing_app_state_t prev_app_state = (ref_app_breathing_app_state_t)0U;
67 
68  printf("Acconeer software version %s\n", acc_version_get());
69 
70  const acc_hal_a121_t *hal = acc_hal_rss_integration_get_implementation();
71 
72  if (!acc_rss_hal_register(hal))
73  {
74  return EXIT_FAILURE;
75  }
76 
77  config = ref_app_breathing_config_create();
78 
79  if (config == NULL)
80  {
81  printf("Failed to create config\n");
82  cleanup(handle, config, sensor, buffer);
83  return EXIT_FAILURE;
84  }
85 
86  set_config(config, DEFAULT_PRESET_CONFIG);
87 
88  handle = ref_app_breathing_create(config);
89 
90  if (handle == NULL)
91  {
92  printf("Failed to create handle\n");
93  cleanup(handle, config, sensor, buffer);
94  return EXIT_FAILURE;
95  }
96 
97  if (!ref_app_breathing_get_buffer_size(handle, &buffer_size))
98  {
99  printf("ref_app_breathing_get_buffer_size() failed\n");
100  cleanup(handle, config, sensor, buffer);
101  return EXIT_FAILURE;
102  }
103 
104  buffer = acc_integration_mem_alloc(buffer_size);
105  if (buffer == NULL)
106  {
107  printf("Failed to allocate buffer\n");
108  cleanup(handle, config, sensor, buffer);
109  return EXIT_FAILURE;
110  }
111 
112  acc_hal_integration_sensor_supply_on(SENSOR_ID);
113  acc_hal_integration_sensor_enable(SENSOR_ID);
114 
115  sensor = acc_sensor_create(SENSOR_ID);
116  if (sensor == NULL)
117  {
118  printf("acc_sensor_create() failed\n");
119  cleanup(handle, config, sensor, buffer);
120  return EXIT_FAILURE;
121  }
122 
123  if (!sensor_calibration(sensor, &sensor_cal_result, buffer, buffer_size))
124  {
125  printf("Sensor calibration failed\n");
126  cleanup(handle, config, sensor, buffer);
127  return EXIT_FAILURE;
128  }
129 
130  if (!ref_app_breathing_prepare(handle, config, sensor, &sensor_cal_result, buffer, buffer_size))
131  {
132  printf("ref_app_breathing_prepare() failed\n");
133  cleanup(handle, config, sensor, buffer);
134  return EXIT_FAILURE;
135  }
136 
137  ref_app_breathing_result_t result = {0};
138 
139  while (true)
140  {
141  if (!measure(sensor, buffer, buffer_size))
142  {
143  cleanup(handle, config, sensor, buffer);
144  return EXIT_FAILURE;
145  }
146 
147  if (!ref_app_breathing_process(handle, buffer, &result))
148  {
149  printf("ref_app_breathing_process() failed\n");
150  cleanup(handle, config, sensor, buffer);
151  return EXIT_FAILURE;
152  }
153 
154  if (!handle_indications(handle, config, sensor, &sensor_cal_result, buffer, buffer_size, &result.presence_result))
155  {
156  cleanup(handle, config, sensor, buffer);
157  return EXIT_FAILURE;
158  }
159 
160  if (!result.presence_result.processing_result.calibration_needed)
161  {
162  print_result(&result, prev_app_state);
163  prev_app_state = result.app_state;
164  }
165  }
166 
167  cleanup(handle, config, sensor, buffer);
168 
169  printf("Application finished OK\n");
170 
171  return EXIT_SUCCESS;
172 }

174 static void cleanup(ref_app_breathing_handle_t *handle, ref_app_breathing_config_t *config, acc_sensor_t *sensor, void *buffer)
175 {
176  acc_hal_integration_sensor_disable(SENSOR_ID);
177  acc_hal_integration_sensor_supply_off(SENSOR_ID);
178 
179  if (config != NULL)
180  {
181  ref_app_breathing_config_destroy(config);
182  }
183 
184  if (sensor != NULL)
185  {
186  acc_sensor_destroy(sensor);
187  }
188 
189  if (buffer != NULL)
190  {
191  acc_integration_mem_free(buffer);
192  }
193 
194  ref_app_breathing_destroy(handle);
195 }

197 static void set_config(ref_app_breathing_config_t *config, breathing_preset_t preset)
198 {
199  acc_detector_presence_config_t *presence_config = config->presence_config;
200 
201  switch (preset)
202  {
203  case BREATHING_PRESET_NONE:
204  // Add custom configuration of the reference application here
205  break;
206  case BREATHING_PRESET_SITTING:
207  acc_detector_presence_config_end_set(presence_config, 1.5f);
208  acc_detector_presence_config_intra_detection_threshold_set(presence_config, 6.0f);
209  break;
210  case BREATHING_PRESET_INFANT:
211  acc_detector_presence_config_end_set(presence_config, 1.0f);
212  acc_detector_presence_config_intra_detection_threshold_set(presence_config, 4.0f);
213  break;
214  }
215 }

217 static bool sensor_calibration(acc_sensor_t *sensor, acc_cal_result_t *sensor_cal_result, void *buffer, uint32_t buffer_size)
218 {
219  bool status = false;
220  bool cal_complete = false;
221  const uint16_t calibration_retries = 1U;
222 
223  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
224  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
225  {
226  // Reset sensor before calibration by disabling/enabling it
227  acc_hal_integration_sensor_disable(SENSOR_ID);
228  acc_hal_integration_sensor_enable(SENSOR_ID);
229 
230  do
231  {
232  status = acc_sensor_calibrate(sensor, &cal_complete, sensor_cal_result, buffer, buffer_size);
233 
234  if (status && !cal_complete)
235  {
236  status = acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS);
237  }
238  } while (status && !cal_complete);
239  }
240 
241  if (status)
242  {
243  /* Reset sensor after calibration by disabling/enabling it */
244  acc_hal_integration_sensor_disable(SENSOR_ID);
245  acc_hal_integration_sensor_enable(SENSOR_ID);
246  }
247  else
248  {
249  printf("acc_sensor_calibrate() failed\n");
250  acc_sensor_status(sensor);
251  }
252 
253  return status;
254 }

256 static bool measure(acc_sensor_t *sensor, void *buffer, uint32_t buffer_size)
257 {
258 
259  if (!acc_sensor_measure(sensor))
260  {
261  printf("acc_sensor_measure failed\n");
262  acc_sensor_status(sensor);
263  return false;
264  }
265 
266  if (!acc_hal_integration_wait_for_sensor_interrupt(SENSOR_ID, SENSOR_TIMEOUT_MS))
267  {
268  printf("Sensor interrupt timeout\n");
269  acc_sensor_status(sensor);
270  return false;
271  }
272 
273  if (!acc_sensor_read(sensor, buffer, buffer_size))
274  {
275  printf("acc_sensor_read() failed\n");
276  acc_sensor_status(sensor);
277  return false;
278  }
279 
280  return true;
281 }

319 static bool handle_indications(ref_app_breathing_handle_t *handle,
320  ref_app_breathing_config_t *config,
321  acc_sensor_t *sensor,
322  acc_cal_result_t *sensor_cal_result,
323  void *buffer,
324  uint32_t buffer_size,
325  acc_detector_presence_result_t *presence_result)
326 {
327  if (presence_result->processing_result.data_saturated)
328  {
329  printf("Data saturated. The detector result is not reliable.\n");
330  }
331 
332  if (presence_result->processing_result.frame_delayed)
333  {
334  printf("Frame delayed. Could not read data fast enough.\n");
335  printf("Try lowering the frame rate or call 'acc_sensor_read' more frequently.\n");
336  }
337 
338  if (presence_result->processing_result.calibration_needed)
339  {
340  printf("Sensor recalibration needed ... \n");
341 
342  if (!sensor_calibration(sensor, sensor_cal_result, buffer, buffer_size))
343  {
344  printf("Sensor calibration failed\n");
345  return false;
346  }
347 
348  printf("Sensor recalibration done!\n");
349 
350  // Before measuring again, the sensor needs to be prepared through the detector
351  if (!ref_app_breathing_prepare(handle, config, sensor, sensor_cal_result, buffer, buffer_size))
352  {
353  printf("acc_detector_presence_prepare() failed\n");
354  return false;
355  }
356  }
357 
358  return true;
359 }
