341 bool ref_app_parking_sensor_calibration(ref_app_parking_handle_t *handle)
342 {
343  bool status = false;
344  bool cal_complete = false;
345  const uint16_t calibration_retries = 1U;
346  const uint32_t calibration_step_timeout_ms = 500U;
347 
348  // Random disturbances may cause the calibration to fail. At failure, retry at least once.
349  for (uint16_t i = 0; !status && (i <= calibration_retries); i++)
350  {
351  // Reset sensor before calibration by disabling/enabling it
352  acc_hal_integration_sensor_disable(handle->sensor_id);
353  acc_hal_integration_sensor_enable(handle->sensor_id);
354 
355  do
356  {
357  status = acc_sensor_calibrate(handle->sensor, &cal_complete, &handle->cal_result, handle->buffer, handle->buffer_size);
358 
359  if (status && !cal_complete)
360  {
361  status = acc_hal_integration_wait_for_sensor_interrupt(handle->sensor_id, calibration_step_timeout_ms);
362  }
363  } while (status && !cal_complete);
364  }
365 
366  if (status)
367  {
368  // Reset sensor after calibration by disabling/enabling it
369  acc_hal_integration_sensor_disable(handle->sensor_id);
370  acc_hal_integration_sensor_enable(handle->sensor_id);
371  }
372 
373  return status;
374 }

376 bool ref_app_parking_noise_calibration(ref_app_parking_handle_t *handle)
377 {
378  // The config to be used for calibration is the same as the 'car config' but with TX off
379  // and a frame rate corresponding to a total measurement time of ~1s.
380  uint16_t noise_subsweep_index = handle->parking_config.car_config.subsweep_index;
381 
382  acc_config_subsweep_enable_tx_set(handle->sensor_config, false, noise_subsweep_index);
383  acc_config_frame_rate_set(handle->sensor_config, (float)CAL_ITERATIONS);
384  handle->measure_timeout_ms = get_measure_timeout_ms((float)CAL_ITERATIONS);
385 
386  if (!acc_sensor_prepare(handle->sensor, handle->sensor_config, &handle->cal_result, handle->buffer, handle->buffer_size))
387  {
388  printf("acc_sensor_prepare() failed\n");
389  return false;
390  }
391 
392  int16_t noise_cal_temps[CAL_ITERATIONS] = {0};
393  float noise_level_sum = 0.0f;
394 
395  uint16_t noise_data_offset = handle->proc_meta.subsweep_data_offset[noise_subsweep_index];
396  uint16_t noise_data_length = handle->proc_meta.subsweep_data_length[noise_subsweep_index];
397 
398  for (uint16_t i = 0; i < CAL_ITERATIONS; i++)
399  {
400  if (!ref_app_parking_measure(handle, false))
401  {
402  return false;
403  }
404 
405  // Only a few measurements and recently calibrated - assume indications ok
406  noise_cal_temps[i] = handle->proc_result.temperature;
407 
408  noise_level_sum += noise_process(&handle->proc_result.frame[noise_data_offset], handle->noise_magnitudes, noise_data_length);
409  }
410 
411  handle->noise_cal_temp = acc_algorithm_median_i16(noise_cal_temps, CAL_ITERATIONS);
412  handle->noise_level = noise_level_sum / (float)CAL_ITERATIONS;
413  handle->noise_level = (handle->noise_level == 0.0f) ? 1.0f : handle->noise_level;
414 
415  // Restore the 'car config'
416  acc_config_subsweep_enable_tx_set(handle->sensor_config, true, noise_subsweep_index);
417  if (handle->parking_config.frame_rate_app_driven)
418  {
419  acc_config_frame_rate_set(handle->sensor_config, 0.0f);
420  }
421  else
422  {
423  acc_config_frame_rate_set(handle->sensor_config, handle->parking_config.frame_rate);
424  }
425 
426  handle->measure_timeout_ms = get_measure_timeout_ms(handle->parking_config.frame_rate);
427 
428  return true;
429 }

431 bool ref_app_parking_obstruction_calibration(ref_app_parking_handle_t *handle)
432 {
433  acc_config_frame_rate_set(handle->sensor_config, (float)CAL_ITERATIONS);
434  handle->measure_timeout_ms = get_measure_timeout_ms((float)CAL_ITERATIONS);
435 
436  uint16_t obs_cal_subsweep_index = handle->parking_config.obstruction_config.subsweep_index;
437  uint16_t obs_cal_data_offset = handle->proc_meta.subsweep_data_offset[obs_cal_subsweep_index];
438  uint16_t obs_cal_data_length = handle->proc_meta.subsweep_data_length[obs_cal_subsweep_index];
439 
440  // Noise calibration without TX
441  acc_config_subsweep_enable_tx_set(handle->sensor_config, false, handle->parking_config.obstruction_config.subsweep_index);
442 
443  if (!acc_sensor_prepare(handle->sensor, handle->sensor_config, &handle->cal_result, handle->buffer, handle->buffer_size))
444  {
445  printf("acc_sensor_prepare() failed\n");
446  return false;
447  }
448 
449  float noise_levels[CAL_ITERATIONS] = {0.0f};
450  float noise_level_sum = 0.0f;
451 
452  for (uint16_t i = 0; i < CAL_ITERATIONS; i++)
453  {
454  if (!ref_app_parking_measure(handle, false))
455  {
456  return false;
457  }
458 
459  // Only a few measurements and recently calibrated - assume indications ok
460 
461  noise_levels[i] = obstruction_noise(&handle->proc_result.frame[obs_cal_data_offset], obs_cal_data_length);
462  noise_level_sum += noise_levels[i];
463  }
464 
465  handle->obstruction_noise_level = noise_level_sum / CAL_ITERATIONS;
466 
467  // Signature calibration with TX - sensitive to environment.
468  // Must be done with sensor free from obstruction and no car present.
469  acc_config_subsweep_enable_tx_set(handle->sensor_config, true, handle->parking_config.obstruction_config.subsweep_index);
470 
471  if (!acc_sensor_prepare(handle->sensor, handle->sensor_config, &handle->cal_result, handle->buffer, handle->buffer_size))
472  {
473  printf("acc_sensor_prepare() failed\n");
474  return false;
475  }
476 
477  signature_t signature = {0};
478  float energy_sum = 0.0f;
479  float weighted_distance_sum = 0.0f;
480 
481  for (uint16_t i = 0; i < CAL_ITERATIONS; i++)
482  {
483  if (!ref_app_parking_measure(handle, false))
484  {
485  return false;
486  }
487 
488  // Only a few measurements and recently calibrated - assume indications ok
489 
490  acc_int16_complex_t *subsweep = &handle->proc_result.frame[obs_cal_data_offset];
491 
492  obstruction_signature(
493  subsweep, handle->obstruction_magnitudes, handle->obstruction_distances, obs_cal_data_length, noise_levels[i], &signature);
494 
495  energy_sum += signature.energy;
496  weighted_distance_sum += signature.weighted_distance;
497  }
498 
499  handle->obstruction_center.energy = energy_sum / CAL_ITERATIONS;
500  handle->obstruction_center.weighted_distance = weighted_distance_sum / CAL_ITERATIONS;
501 
502  handle->obstruction_lp_signature = handle->obstruction_center;
503 
504  // Restore the 'obstruction config'
505  if (handle->parking_config.frame_rate_app_driven)
506  {
507  acc_config_frame_rate_set(handle->sensor_config, 0.0f);
508  }
509  else
510  {
511  acc_config_frame_rate_set(handle->sensor_config, handle->parking_config.frame_rate);
512  }
513 
514  handle->measure_timeout_ms = get_measure_timeout_ms(handle->parking_config.frame_rate);
515 
516  return true;
517 }

608 void ref_app_parking_obstruction_process(ref_app_parking_handle_t *handle, bool *obstruction_detected)
609 {
610  ref_app_parking_config_t *parking_config = &handle->parking_config;
611  acc_config_t *sensor_config = handle->sensor_config;
612  ref_app_parking_obstruction_config_t *obstruction_config = &parking_config->obstruction_config;
613 
614  acc_config_profile_t profile = acc_config_subsweep_profile_get(sensor_config, obstruction_config->subsweep_index);
615 
616  float signal_adjust_factor;
617  float deviation_adjust_factor;
618 
619  acc_processing_get_temperature_adjustment_factors(
620  handle->noise_cal_temp, handle->proc_result.temperature, profile, &signal_adjust_factor, &deviation_adjust_factor);
621 
622  float noise_level_adjusted = handle->obstruction_noise_level * deviation_adjust_factor;
623 
624  acc_int16_complex_t *subsweep = &handle->proc_result.frame[handle->proc_meta.subsweep_data_offset[obstruction_config->subsweep_index]];
625  uint16_t num_points = acc_config_subsweep_num_points_get(sensor_config, obstruction_config->subsweep_index);
626  signature_t curr_signature = {0};
627 
628  // Adjust for temperature changes
629  for (uint16_t i = 0; i < num_points; i++)
630  {
631  subsweep[i].real = ((float)subsweep[i].real / signal_adjust_factor) + 0.5f;
632  subsweep[i].imag = ((float)subsweep[i].imag / signal_adjust_factor) + 0.5f;
633  }
634 
635  // Extract signature
636  obstruction_signature(subsweep, handle->obstruction_magnitudes, handle->obstruction_distances, num_points, noise_level_adjusted, &curr_signature);
637 
638  // Low-pass filter of signature
639  signature_t *lp_signature = &handle->obstruction_lp_signature;
640  float lp_const = handle->obstruction_lp_const;
641 
642  lp_signature->energy = lp_signature->energy * lp_const + (1.0f - lp_const) * curr_signature.energy;
643  lp_signature->weighted_distance = lp_signature->weighted_distance * lp_const + (1.0f - lp_const) * curr_signature.weighted_distance;
644 
645  // Compare against thresholds
646  float energy_diff = fabsf(lp_signature->energy - handle->obstruction_center.energy);
647  float weighted_distance_diff = fabsf(lp_signature->weighted_distance - handle->obstruction_center.weighted_distance);
648 
649  *obstruction_detected =
